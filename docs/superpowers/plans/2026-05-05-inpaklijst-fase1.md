# Inpaklijst — Fase 1 (MVP) Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** A working PWA where Tiemen can create a trip, generate a packing list from his imported library, and check items off (with cross-out for done items and a remove button per item). Single-user, no realtime, no sharing yet.

**Architecture:** React + TypeScript SPA hosted on Netlify, talking to Supabase (Postgres + Auth) via the JS SDK. Schema/RLS deployed via SQL migrations. Excel data imported once via a Node script run locally. UI is mobile-first with Tailwind.

**Tech Stack:** Vite, React 18, TypeScript, Tailwind CSS, Supabase JS, TanStack Query, Zustand, React Router, `xlsx` (Excel import), `vite-plugin-pwa`.

**Spec:** [docs/superpowers/specs/2026-05-05-inpaklijst-design.md](../specs/2026-05-05-inpaklijst-design.md)

**Note on TDD:** This is a UI-heavy app. We TDD the data layer (generator, Excel import) where logic is non-trivial. We skip TDD for purely visual components — instead we verify by running the dev server and clicking through the flow. Each task says explicitly which approach.

-- Helper view: current user's households
create or replace function current_user_households()
returns setof uuid language sql security definer set search_path = public as $$
  select household_id from household_member where user_id = auth.uid();
$$;

create policy household_member_read on household for select
  using (id in (select current_user_households()));
create policy household_owner_write on household for all
  using (id in (
    select household_id from household_member
    where user_id = auth.uid() and role = 'owner'
  ));

create policy hm_self_read on household_member for select
  using (user_id = auth.uid()
      or household_id in (select current_user_households()));
create policy hm_owner_write on household_member for all
  using (household_id in (
    select household_id from household_member
    where user_id = auth.uid() and role = 'owner'

- [ ] **Step 4: Implement generator**

`src/lib/generator.ts`:
```ts
export const UNUSED_FILTER_THRESHOLD = 3;

export type Item = { id: string; name: string; kind: 'packable'|'todo'; default_category: string };
export type Tag = { id: string; name: string; kind: 'triptype'|'weather'|'activity'|'custom' };
export type ItemTag = { item_id: string; tag_id: string };
export type ItemForPerson = { item_id: string; person_id: string };
export type Person = { id: string; name: string };
export type Feedback = {
  item_id: string;
  verdict: 'used'|'unused'|'missing';
  context_snapshot: { triptypes?: string[]; weather?: string[]; activities?: string[] };
};
export type Library = {
  items: Item[];
  tags: Tag[];
  itemTags: ItemTag[];
  itemForPerson: ItemForPerson[];
  persons: Person[];
  feedback: Feedback[];
};
export type Context = {
  persons: string[];
  triptypes: string[];
  weather: string[];
  activities: string[];
};
export type TripItemDraft = { item_id: string; person_id?: string };

export function generateTripItems(lib: Library, ctx: Context): TripItemDraft[] {
  const tagByName = new Map(lib.tags.map(t => [t.name, t]));
  const selectedTagIds = new Set(
    [...ctx.triptypes, ...ctx.weather, ...ctx.activities]
      .map(n => tagByName.get(n)?.id)
      .filter((x): x is string => !!x)
  );
  const tagsByItem = new Map<string, Set<string>>();
  for (const it of lib.itemTags) {
    if (!tagsByItem.has(it.item_id)) tagsByItem.set(it.item_id, new Set());
    tagsByItem.get(it.item_id)!.add(it.tag_id);
  }
  const personsByItem = new Map<string, Set<string>>();
  for (const ip of lib.itemForPerson) {
    if (!personsByItem.has(ip.item_id)) personsByItem.set(ip.item_id, new Set());
    personsByItem.get(ip.item_id)!.add(ip.person_id);
  }

  const drafts: TripItemDraft[] = [];

  for (const item of lib.items) {
    const itemTagIds = tagsByItem.get(item.id) ?? new Set<string>();
    const itemPersonIds = personsByItem.get(item.id) ?? new Set<string>();

    const tagMatch = [...itemTagIds].some(id => selectedTagIds.has(id));
    const personMatch = [...itemPersonIds].some(id => ctx.persons.includes(id));
    const isAlwaysMeename = itemTagIds.size === 0 && itemPersonIds.size === 0;

    if (!(tagMatch || personMatch || isAlwaysMeename)) continue;

    if (countUnusedInContext(lib.feedback, item.id, ctx) >= UNUSED_FILTER_THRESHOLD) continue;

    if (itemPersonIds.size > 0) {
      for (const pid of itemPersonIds) {
        if (ctx.persons.includes(pid)) drafts.push({ item_id: item.id, person_id: pid });
      }
    } else {
      drafts.push({ item_id: item.id });
    }
  }
  return drafts;
}

function countUnusedInContext(feedback: Feedback[], itemId: string, ctx: Context): number {
  return feedback.filter(f =>
    f.item_id === itemId &&
    f.verdict === 'unused' &&
    (f.context_snapshot.triptypes ?? []).some(t => ctx.triptypes.includes(t))
  ).length;
}
```

`scripts/parse-excel.ts`:
```ts
import * as XLSX from 'xlsx';

export type ParsedItem = {
  name: string;
  kind: 'packable' | 'todo';
  default_category: string;
};
export type ParsedItemTag = { item: string; tag: string };
export type ParsedItemPerson = { item: string; person: string };
export type ParsedPerson = { name: string; is_child: boolean };

export type ParsedPlan = {
  persons: ParsedPerson[];
  items: ParsedItem[];
  itemTags: ParsedItemTag[];
  itemForPerson: ParsedItemPerson[];
};

const PERSON_SHEETS = ['AN', 'TIEMEN', 'JAKOB', 'CHARLOTTE'];
const PERSON_CHILD = new Set(['JAKOB', 'CHARLOTTE']);

const CATEGORY_FOR_SHEET: Record<string, { category: string; tags?: string[] }> = {
  STUFF:       { category: 'stuff' },
  ETEN:        { category: 'eten' },
  ELEKTRONICA: { category: 'electronica' },
  PHARMACIE:   { category: 'pharmacie' },
  SPELLETJES:  { category: 'spelletjes' },
  WATERPRET:   { category: 'stuff', tags: ['zwemmen'] },
  FIETS:       { category: 'stuff', tags: ['fiets'] },
  CAMPER:      { category: 'stuff', tags: ['camper'] },
};

export function parseTemplateExcel(filePath: string): ParsedPlan {
  const wb = XLSX.readFile(filePath);

  const persons: ParsedPerson[] = PERSON_SHEETS.map(s => ({
    name: capitalize(s),
    is_child: PERSON_CHILD.has(s),
  }));

  const items: ParsedItem[] = [];
  const itemTags: ParsedItemTag[] = [];
  const itemForPerson: ParsedItemPerson[] = [];
  const seen = new Set<string>();

  // Pass 1: TODO sheet
  for (const sheetName of wb.SheetNames) {
    if (sheetName.trim().toUpperCase() !== 'TODO') continue;
    const ws = wb.Sheets[sheetName];
    const rows = XLSX.utils.sheet_to_json<unknown[]>(ws, { header: 1, defval: '' });
    for (const row of rows) {
      const name = pickName(row);
      if (!name || isHeader(name, 'TO DO')) continue;
      pushItem({ name, kind: 'todo', default_category: 'todo' }, items, seen);
    }
  }

  // Pass 2: category sheets
  for (const sheetName of wb.SheetNames) {
    const upper = sheetName.trim().toUpperCase();
    const cfg = CATEGORY_FOR_SHEET[upper];
    if (!cfg) continue;
    const ws = wb.Sheets[sheetName];
    const rows = XLSX.utils.sheet_to_json<unknown[]>(ws, { header: 1, defval: '' });
    for (const row of rows) {
      const name = pickName(row);
      if (!name || isHeader(name, upper) || isHeader(name, 'WATER') || isHeader(name, 'KAMPEREN')
          || isHeader(name, 'MET FIETS') || isHeader(name, 'ZOMER') || isHeader(name, 'KOLOM1')) continue;
      pushItem({ name, kind: 'packable', default_category: cfg.category }, items, seen);
      for (const tag of cfg.tags ?? []) {
        itemTags.push({ item: name, tag });
      }
    }
  }

  // Pass 3: person sheets
  for (const sheetName of wb.SheetNames) {
    const upper = sheetName.trim().toUpperCase();
    if (!PERSON_SHEETS.includes(upper)) continue;
    const ws = wb.Sheets[sheetName];
    const rows = XLSX.utils.sheet_to_json<unknown[]>(ws, { header: 1, defval: '' });
    for (const row of rows) {
      const name = pickName(row);
      if (!name) continue;
      pushItem({ name, kind: 'packable', default_category: 'kleren' }, items, seen);
      itemForPerson.push({ item: name, person: capitalize(upper) });
    }
  }

  return { persons, items, itemTags, itemForPerson };
}

function pickName(row: unknown[]): string {
  for (const cell of row) {
    if (cell == null) continue;
    const s = String(cell).trim();
    if (s) return s;
  }
  return '';
}
function isHeader(s: string, hdr: string) { return s.trim().toUpperCase() === hdr.toUpperCase(); }
function capitalize(s: string) { return s.charAt(0).toUpperCase() + s.slice(1).toLowerCase(); }
function pushItem(it: ParsedItem, list: ParsedItem[], seen: Set<string>) {
  const key = `${it.kind}::${it.name.toLowerCase()}`;
  if (seen.has(key)) return;
  seen.add(key);
  list.push(it);
}
```

`scripts/import-excel.ts`:
```ts
import { createClient } from '@supabase/supabase-js';
import { parseTemplateExcel } from './parse-excel';
import { resolve } from 'path';

const TEMPLATE = resolve('C:/Users/tieme/Mijn Drive/LEISURE/REIZEN/meenemen op reis/template reis (version 2).xlsx');

async function main() {
  const url = process.env.SUPABASE_URL!;
  const serviceKey = process.env.SUPABASE_SERVICE_ROLE_KEY!;
  const householdId = process.env.HOUSEHOLD_ID!;
  if (!url || !serviceKey || !householdId) {
    console.error('Set SUPABASE_URL, SUPABASE_SERVICE_ROLE_KEY, HOUSEHOLD_ID env vars.');
    process.exit(1);
  }
  const sb = createClient(url, serviceKey);

  console.log(`Parsing ${TEMPLATE}...`);
  const plan = parseTemplateExcel(TEMPLATE);
  console.log(`Found: ${plan.persons.length} persons, ${plan.items.length} items, ` +
              `${plan.itemTags.length} item-tag links, ${plan.itemForPerson.length} item-person links.`);

  console.log('Press Enter to commit, Ctrl-C to abort...');
  await new Promise<void>(r => process.stdin.once('data', () => r()));

  // Seed default tags
  await sb.rpc('seed_default_tags', { h_id: householdId });

  // Insert persons
  const personIdByName = new Map<string, string>();
  for (const p of plan.persons) {
    const { data, error } = await sb.from('person').insert({
      household_id: householdId, name: p.name, is_child: p.is_child,
    }).select('id').single();
    if (error) throw error;
    personIdByName.set(p.name, data.id);
  }

  // Insert items
  const itemIdByKey = new Map<string, string>();
  for (const it of plan.items) {
    const { data, error } = await sb.from('item').insert({
      household_id: householdId, ...it,
    }).select('id').single();
    if (error) throw error;
    itemIdByKey.set(`${it.kind}::${it.name.toLowerCase()}`, data.id);
  }

  // Fetch tags
  const { data: tags } = await sb.from('tag').select('id,name').eq('household_id', householdId);
  const tagIdByName = new Map(tags!.map(t => [t.name, t.id]));

  // item_tag
  for (const it of plan.itemTags) {
    const item_id = itemIdByKey.get(`packable::${it.item.toLowerCase()}`);
    const tag_id = tagIdByName.get(it.tag);
    if (item_id && tag_id) await sb.from('item_tag').insert({ item_id, tag_id });
  }

  // item_for_person
  for (const ip of plan.itemForPerson) {
    const item_id = itemIdByKey.get(`packable::${ip.item.toLowerCase()}`);
    const person_id = personIdByName.get(ip.person);
    if (item_id && person_id) await sb.from('item_for_person').insert({ item_id, person_id });
  }

  console.log('Done.');
}
main().catch(e => { console.error(e); process.exit(1); });
```

import { useState } from 'react';

type Props = {
  name: string;
  checked: boolean;
  addedManually: boolean;
  onToggle: () => void;
  onRemove: () => void;
};

export function ItemRow({ name, checked, addedManually, onToggle, onRemove }: Props) {
  const [confirming, setConfirming] = useState(false);

  function handleRemove() {
    if (addedManually) { onRemove(); return; }
    if (confirming) { onRemove(); return; }
    setConfirming(true);
    setTimeout(() => setConfirming(false), 2500);
  }

  return (
    <div className={`flex items-center gap-3 px-4 py-3 border-b border-zinc-200
                     ${checked ? 'bg-zinc-50' : 'bg-white'}`}>
      <div className={`flex-1 text-[15px]
                       ${checked ? 'line-through text-muted' : 'text-ink'}`}>
        {name}
      </div>

      {/* Green check */}
      <button
        aria-label={checked ? 'Uitvinken' : 'Afvinken'}
        onClick={onToggle}
        className={`w-9 h-9 rounded-md border-2 border-accent flex items-center justify-center
                    text-lg font-semibold transition-colors
                    ${checked ? 'bg-accent text-white' : 'bg-white text-transparent hover:bg-accent/5'}`}
      >✓</button>

      {/* Red cross */}
      <button
        aria-label={confirming ? 'Bevestig verwijderen' : 'Verwijderen uit lijst'}
        onClick={handleRemove}
        className={`w-9 h-9 rounded-md border-2 flex items-center justify-center
                    text-xl font-semibold leading-none transition-colors
                    ${confirming
                      ? 'bg-accent2 border-accent2 text-white'
                      : 'border-accent2 text-accent2 hover:bg-accent2/5'}`}
      >✕</button>
    </div>
  );
}
